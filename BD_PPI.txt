-- BASE DE DATOS PPI

-- TABLA USUARIOS
CREATE TABLE usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    correo VARCHAR(100),
    contrasena VARCHAR(50),
    rol VARCHAR(50)
);

-- TABLA PROVINCIAS
CREATE TABLE provincias (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre_provincia VARCHAR(75)
);

-- TABLA CIUDAD-CANTON
CREATE TABLE ciudad_canton (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre_cc VARCHAR(75),
    provincia_id_fk INT,
    FOREIGN KEY (provincia_id_fk) REFERENCES provincias(id)
);

-- TABLA CATEGORIAS
CREATE TABLE categorias (
    id INT AUTO_INCREMENT PRIMARY KEY,
    categoria_nombre VARCHAR(255)
);


-- TABLA SITIOS
CREATE TABLE sitios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(100),
    descripcion VARCHAR(100),
    contenido TEXT,
    img LONGBLOB DEFAULT NULL,
    estado VARCHAR(50),
    provincias_id_fk INT,
    ciudad_canton_id_fk INT,
    usuarios_id_fk INT,
    categorias_id_fk INT,
    FOREIGN KEY (provincias_id_fk) REFERENCES provincias(id),
    FOREIGN KEY (ciudad_canton_id_fk) REFERENCES ciudad_canton(id),
    FOREIGN KEY (usuarios_id_fk) REFERENCES usuarios(id),
    FOREIGN KEY (categorias_id_fk) REFERENCES categorias(id)
);
